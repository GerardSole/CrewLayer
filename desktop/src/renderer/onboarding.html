Step 2 of 3

Connect your agent

Install the SDK and wire in your API key. Your API key is already pre-filled.

pip install crewlayer from crewlayer import CrewLayerClient client = CrewLayerClient( api_key="crwl_...", base_url="http://localhost:8000" ) # Save a memory client.memory.save( agent_id="your-agent-id", content="User prefers concise answers", importance=0.9 ) # Recall memories memories = client.memory.recall( agent_id="your-agent-id", query="user preferences" )
npm install crewlayer import { CrewLayerClient } from 'crewlayer'; const client = new CrewLayerClient({ apiKey: 'crwl_...', baseUrl: 'http://localhost:8000', }); // Save a memory await client.memory.save({ agentId: 'your-agent-id', content: 'User prefers concise answers', importance: 0.9, }); // Recall memories const memories = await client.memory.recall({ agentId: 'your-agent-id', query: 'user preferences', });
curl -X POST http://localhost:8000/v1/memory/save \ -H "Authorization: Bearer crwl_..." \ -H "Content-Type: application/json" \ -d '{ "agent_id": "your-agent-id", "content": "User prefers concise answers", "importance": 0.9 }'
